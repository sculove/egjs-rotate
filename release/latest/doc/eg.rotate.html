Namespace: rotate

eg.rotate

Tiny custom rotate event binder

Example

var handler = function(e, info){
     info.isVertical;
}
// bind
eg.rotate.on(handler);


// unbind
eg.rotate.off(handler);


// unbind all event attached (call without listener param)
eg.rotate.off();
